# 创建hive表

CREATE EXTERNAL TABLE temp_carinfo partitioned by (day_id string, hr string)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.avro.AvroSerDe' 
STORED AS INPUTFORMAT 'org.apache.hadoop.hive.ql.io.avro.AvroContainerInputFormat'
OUTPUTFORMAT 'org.apache.hadoop.hive.ql.io.avro.AvroContainerOutputFormat' 
LOCATION '/project/pq_auto_analysis_bj_sh/pq_auto_analysis_bj_sh/IDataIn/carInfoAndLocation/mixture/carinfo/topics/pq_auto_analysis_bj_sh__replicator__carInfoAndLocation__autocar' TBLPROPERTIES (
  'avro.schema.literal' = '{
            "type": "record",
            "name": "location",
            "namespace": "com.iflytek.avro",
            "fields": [
                { "name":"log",  "type":"string"}
            ]
       }'
);
  1. 需要定义SERDE
  2. 需要定义avro数据schema
