# hive 常用语句

# 查看表信息

  1. 查看建表语句
show create table tablename;
  1. 查看表分区
show tablename partitions;
  1. 查看表结构
desc tablename;
or 
describe tablename;

# 修改表结构

  1. 修改表名
alter table tablename rename to newtablename;
  1. 增加列
ALTER TABLE table_name  ADD columns (column_name datatype);
  1. 修改列
ALTER TABLE 
    table_name CHANGE [COLUMN] col_old_name col_new_name column_type 
    [COMMENT col_comment] [FIRST|AFTER column_name]

这个命令可以允许改变列名、数据类型、注释、列位置或者它们的任意组合

# 修改表数据

  1. 删除分区
alter table tablename drop if exists partition (partition_name = '${partition_value}') ;
  1. 增加分区
alter table
  tablename
add
  partition (partition_name = '${partition_value}') location 'location_path/${partition_value}';

# 查询表数据

  1. 查询结果写入文件
INSERT OVERWRITE DIRECTORY '/tmp/hdfs_out' SELECT a.* FROM invites a WHERE a.ds='<DATE>';
  1. 查询结果写入表
INSERT OVERWRITE TABLE events SELECT a.* FROM profiles a;
  1. 查询结果插入表
FROM invites a INSERT OVERWRITE TABLE events SELECT a.bar, count(1) WHERE a.foo > 0 GROUP BY a.bar;

# 动态分区

set hive.exec.dynamic.partition.mode=nonstrict;
insert into tmp_ads_meb_rnw_in_m_line_2 partition(month_id)
select open_id,open_name,dev_exp_num,month_id from tmp_ads_meb_rnw_in_m_test_2;
